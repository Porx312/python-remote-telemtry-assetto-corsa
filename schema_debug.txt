CREATE TABLE `lap_records` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `steam_id` varchar(50) DEFAULT NULL,
  `car_model` varchar(100) DEFAULT NULL,
  `track` varchar(100) DEFAULT NULL,
  `server_name` varchar(100) DEFAULT NULL,
  `lap_time` int(11) NOT NULL,
  `date` timestamp NOT NULL DEFAULT current_timestamp(),
  `valid_lap` tinyint(1) DEFAULT 1,
  `timestamp` bigint(20) DEFAULT 0,
  `track_config` varchar(255) DEFAULT '',
  PRIMARY KEY (`id`),
  UNIQUE KEY `unique_lap` (`steam_id`,`car_model`,`track`),
  CONSTRAINT `lap_records_ibfk_1` FOREIGN KEY (`steam_id`) REFERENCES `drivers` (`steam_id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=706 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci
